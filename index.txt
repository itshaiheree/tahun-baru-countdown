const express = require('express')
const app = express()
const port = 9163
const fs = require('fs')
const { marked } = require('marked');
const data = fs.readFileSync('./lp.md', { encoding: 'utf8', flag: 'r' });


const html = marked.parse(data);

app.use(express.static('public'))


app.get('/', (req, res) => {
  res.send(`<style>body { margin: 1 }</style>
<body>
${html}
</body>`)
})

app.listen(port, () => {
  console.log(`Example app listening on port ${port}`)
})
